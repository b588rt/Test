Sub CopyHeader()
    Dim ws4 As Worksheet
    Dim cellB As Range
    Dim lastRow4 As Long
    Dim targetRow As Long
  
    Set ws4 = ThisWorkbook.Sheets("工作表4")
  
    lastRow4 = ws4.Cells(ws4.Rows.Count, "B").End(xlUp).Row
  
    For Each cellB In ws4.Range("B1:B" & lastRow4)
        If Not IsEmpty(cellB) Then
  
            If cellB.Value = "S1" Then
  
                targetRow = cellB.Row + 2
  
                ws4.Rows(1).Copy ws4.Rows(targetRow)
            End If
        End If
    Next cellB
End Sub
