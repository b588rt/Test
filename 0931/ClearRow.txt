Sub ClearRowBlank()
    Dim ws4 As Worksheet
    Dim cellC As Range
    Dim lastRow4 As Long

    Set ws4 = ThisWorkbook.Sheets("工作表4")

    lastRow4 = ws4.Cells(ws4.Rows.Count, "C").End(xlUp).Row

    For Each cellC In ws4.Range("C1:C" & lastRow4)
        If Not IsEmpty(cellC) Then

            If cellC.Value = "VN" Then

                If cellC.Row < ws4.Rows.Count Then 
                    If IsEmpty(ws4.Cells(cellC.Row + 1, "C")) Then

                        ws4.Rows(cellC.Row).ClearContents
                    End If
                End If
            End If
        End If
    Next cellC
End Sub
