Sub CopyAndInsertRows()
    Dim ws1 As Worksheet
    Dim ws2 As Worksheet
    Dim lastRow As Long
    Dim i As Long
    Dim searchValue As Variant
    Dim found As Range
    Dim sourceRow As Long
    Dim targetRow As Long
    
    
    Set ws1 = Worksheets("工作表4")
    Set ws2 = Worksheets("工作表01")
    
    lastRow = ws1.Cells(ws1.Rows.Count, "E").End(xlUp).Row  'check sheet4 E row
    
   
    For i = lastRow To 2 Step -1
        searchValue = ws1.Cells(i, "E").Value
        
       
        If IsEmpty(searchValue) Or searchValue = "" Then   'If E row empty then next
            GoTo NextIteration
        End If
        
        ' check sheet01 same value
        Set found = ws2.Range("C:C").Find(What:=searchValue, LookIn:=xlValues, LookAt:=xlWhole)
        
        If Not found Is Nothing Then
            sourceRow = found.Row
            targetRow = i + 1  
            
            ' copy sheet01 to sheet4
            ws2.Rows(sourceRow).Copy Destination:=ws1.Rows(targetRow)
            
            ' sheet4 insert 1 row
            ws1.Rows(targetRow + 1).Insert Shift:=xlDown
        End If
        
NextIteration:
    Next i
    
   
    Application.CutCopyMode = False
    
   
End Sub
