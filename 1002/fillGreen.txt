

Sub FillGreenAroundMA()
    Dim ws As Worksheet
    Dim lastRow As Long
    Dim r As Long
    
   
    Set ws = ThisWorkbook.Worksheets("工作表4")
    
    lastRow = ws.Cells(ws.Rows.Count, 5).End(xlUp).Row   ' find E row
    
    For r = 1 To lastRow

        If ws.Cells(r, 5).Value = "MA" Then

            ws.Cells(r, 5).Interior.Color = RGB(0, 255, 0)   'MA fill Green

            ws.Cells(r, 4).Interior.Color = RGB(0, 255, 0)   'MA left 1 fill Green

            ws.Cells(r, 6).Interior.Color = RGB(0, 255, 0)   'MA Right 1 fill Green

            ws.Cells(r, 7).Interior.Color = RGB(0, 255, 0)   'MA Right 2 fill Green

        End If
    Next r
    
End Sub
