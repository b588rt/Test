Sub CopyBelowNI()

    Dim ws As Worksheet
    Dim lastRow As Long
    Dim r As Long
    Dim j As Long
    
  
    Set ws = ThisWorkbook.Worksheets("工作表4")
    
  
    lastRow = ws.Cells(ws.Rows.Count, 7).End(xlUp).Row  ' G Row
    
    For r = 1 To lastRow

        If ws.Cells(r, 7).Value = "NI" Then
            j = r + 1  ' NI next row
            Dim nextRow As Long
            nextRow = r + 2  
            If nextRow <= ws.Rows.Count Then

                ws.Cells(nextRow, 7).Value = ws.Cells(j, 7).Value
            End If
        End If
    Next r
    
End Sub
