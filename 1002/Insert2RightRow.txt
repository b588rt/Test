Sub InsertCellsBelowSE()
    Dim ws As Worksheet
    Dim lastRow As Long
    Dim r As Long
    Dim j As Long
    
   
    Set ws = ThisWorkbook.Worksheets("工作表4")
    
   
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    
   
    For r = 1 To lastRow
   
        If ws.Cells(r, 1).Value = "SE" Then
            j = r + 2 

            If j <= ws.Rows.Count Then

                ws.Range(ws.Cells(j, 2), ws.Cells(j, 4)).Insert Shift:=xlToRight  'insert 4 row
            End If
        End If
    Next r
    

End Sub
